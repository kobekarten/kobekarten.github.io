% Handout style: colored footer on all pages, meta header on p.1

\definecolor{handoutline}{HTML}{b5cbd6}

% --- Base page style: only footer, no header rule ---------
\pagestyle{fancy}
\fancyhf{}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

\fancyfoot[C]{%
  \parbox{\linewidth}{\centering
    \textcolor{handoutline}{\rule{\linewidth}{0.4pt}}\\[3pt]%
    \textcolor{handoutline}{\small $for(author)$$author$$sep$, $endfor$$if(email)$ | $email$$endif$$if(date)$ | $date$$endif$}%
  }%
}

% --- First-page style: meta-info left, logo right ---------
\fancypagestyle{firstpage}{%
  \fancyhf{}%
  \renewcommand{\headrulewidth}{0.4pt}%
  \renewcommand{\headrule}{\vskip 6pt{\color{handoutline}\hrule width\headwidth height\headrulewidth \vss}}%
  \renewcommand{\footrulewidth}{0pt}%
  \fancyhead[L]{%
    \parbox[t]{0.6\linewidth}{\raggedright\small
      $if(organization)$$organization$\\$endif$%
      $if(institute)$$institute$\\$endif$%
      $if(course)$$course$\\$endif$%
      $if(term)$$term$$endif$%
    }%
  }%
  \fancyhead[R]{%
    \raisebox{-\handoutlogooffset}{\includegraphics[height=1.2cm]{logo.png}}%
  }%
  \fancyfoot[C]{%
    \parbox{\linewidth}{\centering
      \textcolor{handoutline}{\rule{\linewidth}{0.4pt}}\\[3pt]%
      \textcolor{handoutline}{\small $for(author)$$author$$sep$, $endfor$$if(email)$ | $email$$endif$$if(date)$ | $date$$endif$}%

  }%
}
